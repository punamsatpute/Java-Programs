// For Loop console.log("Using for loop:"); for (let i = 1; i <= 5; i++) { console.log(i); } // For-In Loop (Iterating through an array) console.log("\nUsing for-in loop:"); const numbers = [1, 2, 3, 4, 5]; for (let index in numbers) { console.log(numbers[index]); } // While Loop console.log("\nUsing while loop:"); let count = 1; while (count <= 5) { console.log(count); count++; } // Do-While Loop console.log("\nUsing do-while loop:"); let num = 1; do { console.log(num); num++; } while (num <= 5);
